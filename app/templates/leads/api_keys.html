Kullanım Örnekleri:
Web Sitesi Formu (JavaScript) 
fetch('{{ request.url_root }}api/v1/leads', {
  method: 'POST',
  headers: {
    'X-API-Key': 'YOUR_API_KEY_HERE',
    'Content-Type': 'application/json'
  },
  body: JSON.stringify({
    source: 'website',
    first_name: 'Ahmet',
    last_name: 'Yılmaz',
    email: 'ahmet@example.com',
    phone: '+90 555 123 4567',
    interested_service: 'hair',
    message: 'Saç ekimi hakkında bilgi almak istiyorum'
  })
})
.then(response => response.json())
.then(data => {
  console.log('Lead oluşturuldu:', data);
  alert('Başvurunuz alındı!');
})
.catch(error => console.error('Hata:', error));
